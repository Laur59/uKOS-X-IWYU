cmake_minimum_required(VERSION 3.26)

project(FatFs_All
    DESCRIPTION "FatFs static libraries for all supported cores"
    LANGUAGES NONE
)

message("╔════════════════════════════════════════════════════════════╗")
message("║           Building FatFs for All Architectures             ║")
message("╚════════════════════════════════════════════════════════════╝")
message("")

# List of all supported cores
set(SUPPORTED_CORES
    CORTEX_M3
    CORTEX_M33
    CORTEX_M4
    CORTEX_M55
    CORTEX_M7
    RV32IMAC
    RV64IMAFDC
)

# Create a custom target for each core that builds independently
foreach(CORE ${SUPPORTED_CORES})
    if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${CORE}/CMakeLists.txt")
        message(STATUS "Configuring build for core: ${CORE}")

        # Create a custom target that builds this core independently
        add_custom_target(build_${CORE}
            COMMAND ${CMAKE_COMMAND} -B ${CMAKE_CURRENT_SOURCE_DIR}/${CORE}/build -S ${CMAKE_CURRENT_SOURCE_DIR}/${CORE}
            COMMAND ${CMAKE_COMMAND} --build ${CMAKE_CURRENT_SOURCE_DIR}/${CORE}/build
            WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/${CORE}
            COMMENT "Building FatFs for ${CORE}"
            VERBATIM
        )

        list(APPEND ALL_CORE_TARGETS build_${CORE})
    else()
        message(WARNING "Skipping ${CORE}: CMakeLists.txt not found")
    endif()
endforeach()

# Create a master target that builds all cores
# The ALL keyword makes this the default target
add_custom_target(build_all_cores ALL
    DEPENDS ${ALL_CORE_TARGETS}
    COMMENT "Building FatFs for all architectures"
)

# Summary message
message("")
message("──────────────────────────────────────────────────────────────")
message("Configuration complete.")
message("")
message("To build all libraries, run:")
message("  cmake --build build --target build_all_cores")
message("")
message("To build a specific core, run:")
message("  cmake --build build --target build_CORTEX_M4")
message("──────────────────────────────────────────────────────────────")
